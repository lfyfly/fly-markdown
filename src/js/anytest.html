面向对象

属性可以传参数

多个实例对象的方法指向同一个内存

传统实现方法1

function Person(name,age){
    this.name=name;
    this.age=age;
}
Person.prototype={
    constructor:Person,   //因为重写需要矫正constructor
    getName:function(){
        console.log(this.name)
    },
    getAge: function(){
        console.log(this.age)
    }
}
var person1=new Person('lin','18');
console.log(person1.name) //'lin'
person1.getAge();//18

继承

==原型链==

通过对象的属性 _proto\_ 属性指向一个对象来继承

注意: ==引用类型属性的修改问题==

构造函数之间的继承

function Man(name,age,sex){
    Person.call(this,name,age); //继承父类的属性且可以传参数 
    this.sex=sex;

    Man.prototype.getSex=function(){
        console.log(this.sex)
    }
}
Man.prototype =Object.create(Person.prototype);//>=ie9
//Man.prototype.__proto =Person.prototype 

var man1 = new Man('lin',18,'boy');
console.log(man1.sex) //boy
console.log(man1.name) //lin
man1.getName() //lin
man1.getSex()  //boy

es6继承

类之间的继承

class Man extends Person{
    constructor(name,age,sex){
        super(name,age)
        this.sex=sex
    }
    getSex(){
        console.log(this.sex)
    }
}

var man1 = new Man('lin',18,'boy');
console.log(man1.sex) //boy
console.log(man1.name) //lin
man1.getName() //lin
man1.getSex()  //boy

==关于super==

super这个关键字，既可以当作函数使用，也可以当作对象使用。

  • ==super作为函数调用时，代表父类的构造函数Parent(super内部的this指的是子类)==
  • super()作为函数只能用在子类的构造函数之中调用，用在其他地方就会报错

super作为对象时，指向父类的原型对象Parent.prototype。

  • super在子类内调用方法时，方法类==this指向子类实例对象==

对象深拷贝

==错误深拷贝==

var copy ={}
copy=Object.create(Object.getPrototypeOf(man1));// >=ie9
//copy.__proto__=man1.__proto__

for(var x in man1){ //错误：x包括自身属性和prototype属性（但是不包括不可枚举属性）
    copy[x]=man1[x]
}

console.log(copy.sex) //boy
console.log(copy.name) //lin
copy.getName() //lin
copy.getSex()  //boy

==完全深拷贝==（不可枚举属性，属性描述）

function copy(obj) {
  var copy = Object.create(Object.getPrototypeOf(obj));
  //copy.__proto__===obj.__proto__

  var propNames = Object.getOwnPropertyNames(obj);
    //obj对象上的所有属性（包括不可枚举属性）（不包括prototype的属性）
  propNames.forEach(function(name) {
    var desc = Object.getOwnPropertyDescriptor(obj, name);//获取属性描述，里面有属性值
    Object.defineProperty(copy, name, desc);
  });

  return copy;
}

var obj1 = { a: 1, b: 2 };
var obj2 = copy(obj1); // obj2 looks like obj1 now
